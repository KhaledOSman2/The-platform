    <script>
        // دالة استخراج معرف يوتيوب
        function getYoutubeId(url) {
            const regex = /(?:youtube(?:-nocookie)?\.com\/(?:[^\/\n\s]+\/\S+\/|(?:v|e(?:mbed)?)\/|.*[?&]v=)|youtu\.be\/)([a-zA-Z0-9_-]{11})/;
            const match = url.match(regex);
            if (match) {
                console.log("تم استخراج معرف الفيديو:", match[1]);
            } else {
                console.log("لم يتم استخراج معرف الفيديو من الرابط:", url);
            }
            return match ? match[1] : null;
        }

        let player; // مشغل Plyr

        document.addEventListener('DOMContentLoaded', function () {
            const token = localStorage.getItem('token');
            if (!token) {
                window.location.href = 'login.html';
                return;
            }

            // دالة تهيئة المشغل مع كافة الخيارات والتحديثات
            function initializePlayer(videoId) {
                const container = document.getElementById('courseVideoContainer');
                // تعيين السمات المطلوبة للمشغل (لـ YouTube)
                container.setAttribute('data-plyr-provider', 'youtube');
                container.setAttribute('data-plyr-embed-id', videoId);
                // إذا كان مشغل سابق موجود، يتم تدميره
                if (player) {
                    player.destroy();
                }
                // تهيئة مشغل Plyr مع كافة الخيارات المطلوبة
                player = new Plyr(container, {
                    controls: [
                        'play-large', 'play',
                        'progress', 'current-time', 'mute',
                        'captions', 'settings', 'fullscreen'
                    ],
                    settings: ['quality', 'speed', 'loop'],
                    autoplay: false,
                    muted: false,
                    loop: { active: false },
                    ratio: '16:9',
                    youtube: {
                        noCookie: false,
                        customVars: {
                            autoplay: 1,
                            modestbranding: 1,
                            rel: 0,
                            controls: 0,
                            iv_load_policy: 3
                        }
                    },
                    i18n: {
                        play: 'تشغيل',
                        pause: 'إيقاف',
                        seek: 'تقديم/رجوع',
                        fullscreen: 'تكبير الشاشة',
                        exitFullscreen: 'خروج من وضع التكبير',
                        settings: 'الإعدادات',
                        speed: 'السرعة',
                        quality: 'الجودة',
                        loop: 'تكرار'
                    },
                    keyboard: {
                        focused: true,
                        global: false
                    },
                    storage: {
                        enabled: true,
                        key: 'plyr'
                    },
                    tooltips: {
                        controls: true,
                        seek: true
                    }
                });

                // عند جاهزية المشغل، تعديل سمات الـ iframe لإخفاء علامات YouTube الإضافية
                player.on('ready', event => {
                    const iframe = player.embed.getIframe();
                    if (iframe) {
                        iframe.setAttribute('allowfullscreen', '');
                        iframe.setAttribute('allowtransparency', '');
                        iframe.setAttribute('allow', 'autoplay; encrypted-media; fullscreen; picture-in-picture');
                    }
                });
            }

            // استرجاع معرف الكورس من معلمات الرابط
            const urlParams = new URLSearchParams(window.location.search);
            const courseId = urlParams.get('id');

            if (courseId) {
                fetch(`/api/courses/${courseId}`, {
                    headers: { Authorization: `Bearer ${token}` }
                })
                    .then(response => {
                        if (!response.ok) {
                            throw new Error('Network response was not ok');
                        }
                        return response.json();
                    })
                    .then(course => {
                        // تعبئة تفاصيل الكورس
                        document.getElementById('courseTitle').textContent = course.title;
                        document.getElementById('courseGrade').textContent = course.grade;
                        document.getElementById('courseImage').src = course.imageURL;

                        if (course.videos && course.videos.length) {
                            const firstVideoUrl = course.videos[0].url || course.videoURL;
                            const videoId = getYoutubeId(firstVideoUrl);
                            if (videoId) {
                                initializePlayer(videoId);
                            }
                            document.getElementById('videoTitle').textContent = course.videos[0]?.title || '';

                            // تعبئة قائمة الفيديوهات
                            const videosList = document.getElementById('videosList');
                            videosList.innerHTML = '';
                            course.videos.forEach((video, index) => {
                                const li = document.createElement('li');
                                li.className = 'list-group-item d-flex justify-content-between align-items-center';
                                li.innerHTML = `<div>
                                  <strong>المحاضرة ${index + 1}:</strong> ${video.title}
                                </div>`;
                                li.addEventListener('click', function () {
                                    const newVideoId = getYoutubeId(video.url);
                                    if (newVideoId) {
                                        initializePlayer(newVideoId);
                                    }
                                    document.getElementById('videoTitle').textContent = video.title;
                                });
                                videosList.appendChild(li);
                            });
                            document.getElementById('noVideoMessage').style.display = 'none';
                        } else {
                            document.getElementById('courseVideoContainer').style.display = 'none';
                            document.getElementById('noVideoMessage').style.display = 'block';
                            document.getElementById('videosList').innerHTML = '<li class="list-group-item">لا تتوفر فيديوهات حاليا</li>';
                        }

                        // تعبئة قائمة المستندات
                        const activitiesList = document.getElementById('activitiesList');
                        activitiesList.innerHTML = '';
                        if (course.activities && course.activities.length) {
                            course.activities.forEach((activity, index) => {
                                const li = document.createElement('li');
                                li.className = 'list-group-item d-flex justify-content-between align-items-center';
                                li.innerHTML = `<div>
                                  <strong>المستند ${index + 1}:</strong> ${activity.title}
                                </div>
                                <a href="${activity.filePath}" class="btn btn-primary" download>تحميل</a>`;
                                activitiesList.appendChild(li);
                            });
                        } else {
                            activitiesList.innerHTML = '<li class="list-group-item">لا تتوفر مستندات حاليا</li>';
                        }

                        // تعبئة قائمة الاختبارات والواجبات
                        const examsList = document.getElementById('examsList');
                        examsList.innerHTML = '';
                        fetch(`/api/exams?courseId=${courseId}&grade=${course.grade}`, {
                            headers: { Authorization: `Bearer ${token}` }
                        })
                            .then(response => {
                                if (!response.ok) {
                                    throw new Error('Network response was not ok');
                                }
                                return response.json();
                            })
                            .then(({ exams }) => {
                                if (exams && exams.length) {
                                    exams.forEach((exam, index) => {
                                        const li = document.createElement('li');
                                        li.className = 'list-group-item d-flex justify-content-between align-items-center';
                                        li.innerHTML = `<div>
                                      <strong>الاختبار ${index + 1}:</strong> ${exam.title}
                                    </div>
                                    <button class="btn btn-primary" onclick="viewExam('${exam.googleFormUrl}')">عرض</button>`;
                                        examsList.appendChild(li);
                                    });
                                } else {
                                    examsList.innerHTML = '<li class="list-group-item">لا تتوفر اختبارات او واجبات حاليا</li>';
                                }
                            })
                            .catch(err => {
                                console.error('حدث خطأ أثناء جلب بيانات الامتحانات:', err);
                                alert('انتهت صالحية الجلسة برجاء اعادة تسجيل الدخول');
                            });
                    })
                    .catch(err => {
                        console.error('حدث خطأ أثناء جلب بيانات الكورس:', err);
                        alert('حدث خطأ أثناء جلب بيانات الكورس');
                    });
            } else {
                alert('لم يتم العثور على معرف الكورس');
            }

            // تحديث روابط التنقل
            const navLinks = document.getElementById('navLinks');
            navLinks.innerHTML = `
        <li class="nav-item"><a class="nav-link" href="index.html">الرئيسية</a></li>
        <li class="nav-item"><a class="nav-link" href="courses.html">الكورسات</a></li>
        <li class="nav-item"><a class="nav-link" href="dashboard.html">لوحة التحكم</a></li>
        <li class="nav-item"><a class="nav-link" href="#" id="logoutLink">تسجيل خروج</a></li>
      `;
            document.getElementById('logoutLink').addEventListener('click', function (e) {
                e.preventDefault();
                localStorage.removeItem('token');
                window.location.href = 'login.html';
            });

            // وظيفة الطي للبطاقات
            document.querySelectorAll('.card-header').forEach(header => {
                header.addEventListener('click', function () {
                    const cardBody = this.nextElementSibling;
                    cardBody.classList.toggle('show');
                });
            });
        });

        function viewExam(googleFormUrl) {
            const examModal = new bootstrap.Modal(document.getElementById('examModal'));
            document.getElementById('examIframe').src = googleFormUrl;
            examModal.show();
        }
    </script>

    <script>
        window.onscroll = function () {
            scrollFunction();
        };

        function scrollFunction() {
            const backToTopBtn = document.getElementById("backToTopBtn");
            if (document.body.scrollTop > 20 || document.documentElement.scrollTop > 20) {
                backToTopBtn.style.display = "block";
            } else {
                backToTopBtn.style.display = "none";
            }
        }

        function topFunction() {
            document.body.scrollTop = 0;
            document.documentElement.scrollTop = 0;
        }
    </script>
